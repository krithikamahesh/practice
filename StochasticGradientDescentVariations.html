def StochasticGD(x,y,theta,learn_rate,max_iter):
    i = 0
    
    a = np.ones(10000)
    b = np.array(x)
    # 2-d array of (1,x) pairs
    x2d = np.column_stack((a,b))
    
    #lists to keep track of parameters
    stoc_loss = []
    grad_x = []
    grad_y = []
    theta_x = []
    theta_y = []
    
    #stop criterion: difference in consecutive losses is zero (rounded to 6 decimal places)
    loss_prev = 1
    loss_val = 0
    gradient = np.array([0,0])
        
    xy = list(zip(x,y))
    #stop criterion: difference in consecutive losses is zero (rounded to 6 decimal places)
    while i <= max_iter and (round(loss_prev,6) - round(loss_val,6) != 0):
        
        tot_sq_error = 0
        np.random.shuffle(xy)
        #Compute parameters for each pair of (x,y)
        for x,y in xy:
            
            sq_error = (theta[0] + (theta[1] * x) - y) ** 2
            tot_sq_error += sq_error
            
            gradient[0] = theta[0] + (theta[1] * x) - y
            gradient[1] = x * gradient[0]
            
            grad_x.append(gradient[1]) 
            grad_y.append(gradient[0])
            theta = theta - learn_rate * gradient
            
            theta_x.append(theta[1])
            theta_y.append(theta[0])
            
        loss_prev = loss_val
        #Mean Square Error (Loss)
        loss_val = 0.0001 * tot_sq_error
        stoc_loss.append(loss_val)
        
        i += 1
    
    print('\nParameters from Stochastic Gradient Descent:\n')
    print('Loss = ',loss_val,'\nIntercept = ',theta[0],'Slope = ',theta[1])
    #Loss versus Iterations
    plt.plot(range(len(stoc_loss)),stoc_loss)
    plt.xlabel('Iterations')
    plt.ylabel('Mean Squared Error/ Loss')
    plt.title('Loss versus Iterations')
    plt.show()
    #Plot of path taken by gradient
    plt.plot(grad_y,grad_x,lw=0.1)
    plt.xlabel('Gradient[1]: Slope Part')
    plt.ylabel('Gradient[0]: Intercept Part')
    plt.title('Path taken by gradient')
    plt.show()

def ExtractMB(x,y,theta,learn_rate,max_iter):
    xy = list(zip(x,y))
    
    #stopping criterion: when the difference between consecutive average losses (averaged over a mini-batch) is zero
    #when rounded to 6 decimal places
    avg_loss_prev = 1
    avg_loss = 0
    
    #lists to keep track of parameters
    avg_loss_arr =[]
    grad_avg_x = []
    grad_avg_y = []
    
    #lists to hold ALL gradient positions encountered
    gX = []
    gY = []
    
    gradient_avg = np.array([0,0])
    theta_avg = np.array([0,0])
    list_of_batches = []
    #number of mini-batches is 100; each with 100 points [(x,y) pairs]
    num_mb = 100
    i = 0
    while i < 100:
        #slice positions
        f = i * 100
        l = f + 100
        list_of_batches.append(xy[f:l])
        i += 1
    j = 0
    #Iterating over all mini-batches:
    while j <= max_iter and (round(avg_loss_prev,6) - round(avg_loss,6) != 0):
        np.random.shuffle(list_of_batches)
        loss = []
        grad_x = []
        grad_y = []
        theta_x = []
        theta_y = []
        k = 0
        
        #For each mini-batch: Compute Parameters
        while k < len(list_of_batches):
            
            a = np.ones(100)
            x = list(zip(*list_of_batches[k]))[0]
            b = np.array(x)
            y = list(zip(*list_of_batches[k]))[1]
            x2d = np.column_stack((a,b))
            
            loss_val = 0
            
            
            y_pred = np.array([])
            error = np.array([])
            
            y_pred = np.dot(x2d,theta)
            error = y_pred - y
             
            loss_val = ((y_pred - y) ** 2).mean(axis=None)
            loss.append(loss_val)
            
            gradient = np.dot(x2d.T, (np.dot(x2d, theta.T) - y))
            
            grad_x.append(gradient[1]) 
            grad_y.append(gradient[0])
            gX.append(gradient[1])
            gY.append(gradient[0])
            
            theta = theta - learn_rate * gradient
            
            theta_x.append(theta[1])
            theta_y.append(theta[0])
            
            k += 1
            
        avg_loss_prev = avg_loss
        avg_loss = sum(loss)/ len(loss)
        avg_loss_arr.append(avg_loss)
        
        gradient_avg[1] = sum(grad_x)/len(grad_x)
        grad_avg_x.append(gradient_avg[1])
        #gX.append(gradient_avg[1])
        gradient_avg[0] = sum(grad_y)/len(grad_y)
        grad_avg_y.append(gradient_avg[0])
        #gY.append(gradient_avg[0])
        
        j += 1
    
    print('\nParameters calculated by gradient descent at iteration ',j)
    print('Avg loss = ',avg_loss,'Intercept = ',sum(theta_y)/100,'Slope = ',sum(theta_x)/100)
    
    
    #Plot of loss versus iterations
    
    plt.plot(range(len(avg_loss_arr)),avg_loss_arr)
    plt.xlabel('Iterations')
    plt.ylabel('Batch Loss')
    plt.title('Loss versus Iterations')
    plt.show()

    plt.plot(gX,gY,lw = 0.1)
    plt.xlabel('Gradient[1]: Slope Part')
    plt.ylabel('Gradient[0]: Intercept Part')

Question 9. Inspecting your input data, and its partitioning; answer the following questions:

• What is the best way to partition your data, given your cluster configuration? How many partitions did you chose?

In the given cluster configuration, there are four cores. By default, the number of partitions is the number of all available cores. Since we did not specify the number of partitions while calling the parallelize function, by default four partitions will be created. Considering our dataset, which is small to be parallelized; a better way would be to use a smaller number of larger sized partitions (2 partitions or even just one partition on 1 core would suffice our data) thereby enabling us to reduce the overhead of data transfer and to make efficient usage of resources.

In general however, with large datasets, a few partitions of a large size would enable quick completion of work provided all workers are busy. However, here, there would be some unused resources. For better distribution of work it is advantageous to use many small partitions; however this would add an overhead for data exchange. Between the above two methods, the first one (a few large partitions) would have a better impact on performance.

Question 10. Finding bottlenecks.

• What is the bottleneck in the current implementation of your algorithm? • Can you nail down exactly which method is causing this bottleneck? • Can you suggest ways to mitigate or eliminate the bottleneck?

The implementation here for distributed Mini Batch SGD is not restricted by a bottleneck due to the small size of the data set. However, the network will be the limiting factor once the sample size increases. Bottleneck-issues increase exponentially with increasing sample size.

Another bottleneck is created through synchronization. If worker 1 has not yet returned its results, workers 2, 3 and 4 must wait in line to submit their results.

When synchronization is the cause of the bottleneck: The method that executes in parallel is not treated as completed until each worker has submitted its results. This implies that certain workers take longer to compute the instructions in the method. (In our example of distributed Mini Batch SGD, with large data sets; the method called 'computeParams' would cause workers to wait such that synchronization is achieved.) One way to mitigate the bottleneck is to switch to an asynchronous distribution.
